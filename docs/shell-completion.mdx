# Shell Completion

Shell completion is available for bash, zsh and fish.

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

<Tabs
    defaultValue="bash"
    values={[
        { label: 'bash', value: 'bash', },
        { label: 'zsh', value: 'zsh', },
        { label: 'fish', value: 'fish', },
    ]}
>

<TabItem value="bash">

```bash
sake completion bash
```

</TabItem>

<TabItem value='zsh'>

```bash
sake completion zsh
```

</TabItem>

<TabItem value='fish'>

```bash
sake completion fish
```

</TabItem>

</Tabs>
